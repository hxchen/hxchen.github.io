$(document).ready(function(){
            var url = "http://hostname/model-art-code";
            $.ajax({
                type : "GET",
                async: false,
                url : url,
                dataType: 'jsonp',
                jsonp: "callback",
                jsonpCallback:"success_jsonpCallback",
                success : function(json){
                    console.log('success='+json.code);
                },
                error:function(err){
                    console.log('fail'+err);
                }
            });
        });
